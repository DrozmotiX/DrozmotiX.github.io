  password: "xxxxxx"

api:
    password: 'MyPassword'

i2c:
  sda: 21
  scl: 22
  scan: True
  id: bus_a

sensor:

    name: "Hal_Illuminance"
    address: 0x23
    measurement_time: 69
    update_interval: 10s
    
# Example configuration for an GPIO output    
output:
  - platform: gpio
    pin: 12
    inverted: true
    id: gpio_12

    name: "Generic Output"
    output: 'gpio_12'
